68  return;
69  }
70  dev.openTag(SUMO_TAG_VTYPE);
71  dev.writeAttr(SUMO_ATTR_ID, id);
72  if (wasSet(VTYPEPARS_LENGTH_SET)) {
73  dev.writeAttr(SUMO_ATTR_LENGTH, length);
74  }
75  if (wasSet(VTYPEPARS_MINGAP_SET)) {
76  dev.writeAttr(SUMO_ATTR_MINGAP, minGap);
77  }
78  if (wasSet(VTYPEPARS_MAXSPEED_SET)) {
79  dev.writeAttr(SUMO_ATTR_MAXSPEED, maxSpeed);
80  }
81  if (wasSet(VTYPEPARS_PROBABILITY_SET)) {
82  dev.writeAttr(SUMO_ATTR_PROB, defaultProbability);
83  }
84  if (wasSet(VTYPEPARS_SPEEDFACTOR_SET)) {
85  dev.writeAttr(SUMO_ATTR_SPEEDFACTOR, speedFactor);
86  }
87  if (wasSet(VTYPEPARS_SPEEDDEVIATION_SET)) {
88  dev.writeAttr(SUMO_ATTR_SPEEDDEV, speedDev);
89  }
90  if (wasSet(VTYPEPARS_VEHICLECLASS_SET)) {
91  dev.writeAttr(SUMO_ATTR_VCLASS, toString(vehicleClass));
92  }
93  if (wasSet(VTYPEPARS_EMISSIONCLASS_SET)) {
94  dev.writeAttr(SUMO_ATTR_EMISSIONCLASS, getVehicleEmissionTypeName(emissionClass));
95  }
96  if (wasSet(VTYPEPARS_IMPATIENCE_SET)) {
97  if (impatience == -std::numeric_limits<SUMOReal>::max()) {
98  dev.writeAttr(SUMO_ATTR_IMPATIENCE, "off");
99  } else {
100  dev.writeAttr(SUMO_ATTR_IMPATIENCE, impatience);
101  }
102  }
103  if (wasSet(VTYPEPARS_SHAPE_SET)) {
104  dev.writeAttr(SUMO_ATTR_GUISHAPE, getVehicleShapeName(shape));
105  }
106  if (wasSet(VTYPEPARS_WIDTH_SET)) {
107  dev.writeAttr(SUMO_ATTR_WIDTH, width);
108  }
109  if (wasSet(VTYPEPARS_HEIGHT_SET)) {
110  dev.writeAttr(SUMO_ATTR_HEIGHT, height);
111  }
112  if (wasSet(VTYPEPARS_COLOR_SET)) {
113  dev.writeAttr(SUMO_ATTR_COLOR, color);
114  }
115  if (wasSet(VTYPEPARS_OSGFILE_SET)) {
116  dev.writeAttr(SUMO_ATTR_OSGFILE, osgFile);
117  }
118  if (wasSet(VTYPEPARS_IMGFILE_SET)) {
119  dev.writeAttr(SUMO_ATTR_IMGFILE, imgFile);
120  }
121 
122  if (cfParameter.size() != 0) {
123  dev.openTag(cfModel);
124  std::vector<SumoXMLAttr> attrs;
125  for (CFParams::const_iterator i = cfParameter.begin(); i != cfParameter.end(); ++i) {
126  attrs.push_back(i->first);
127  }
128  std::sort(attrs.begin(), attrs.end());
129  for (std::vector<SumoXMLAttr>::const_iterator i = attrs.begin(); i != attrs.end(); ++i) {
130  dev.writeAttr(*i, cfParameter.find(*i)->second);
131  }
132  dev.closeTag();
133  dev.closeTag();
134  } else {
135  dev.closeTag();
136  }
137 }

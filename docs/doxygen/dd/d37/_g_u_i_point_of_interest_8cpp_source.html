110 
111  if (myImgFile != "") {
112  int textureID = GUITexturesHelper::getTextureID(myImgFile);
113  if (textureID > 0) {
114  GUITexturesHelper::drawTexturedBox(textureID,
115  -myHalfImgWidth * s.poiExaggeration, -myHalfImgHeight * s.poiExaggeration,
116  myHalfImgWidth * s.poiExaggeration, myHalfImgHeight * s.poiExaggeration);
117  }
118  } else {
119  // fallback if no image is defined
120  GLHelper::drawFilledCircle((SUMOReal) 1.3 * s.poiExaggeration, 16);
121  }
122  glPopMatrix();
123  drawName(Position(x() + 1.32 * s.poiExaggeration, y() + 1.32 * s.poiExaggeration),
124  s.scale, s.poiName);

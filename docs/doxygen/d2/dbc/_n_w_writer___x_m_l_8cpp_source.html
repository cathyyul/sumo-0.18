61 NWWriter_XML::writeNetwork(const OptionsCont& oc, NBNetBuilder& nb) {
62  // check whether plain-output files shall be generated
63  if (oc.isSet("plain-output-prefix")) {
64  writeNodes(oc, nb.getNodeCont());
65  writeEdgesAndConnections(oc, nb.getNodeCont(), nb.getEdgeCont());
66  writeTrafficLights(oc, nb.getTLLogicCont(), nb.getEdgeCont());
67  }
68  if (oc.isSet("junctions.join-output")) {
69  writeJoinedJunctions(oc, nb.getNodeCont());
70  }
71  if (oc.isSet("street-sign-output")) {
72  writeStreetSigns(oc, nb.getEdgeCont());
73  }
74 }

77 void
78 NWWriter_XML::writeNodes(const OptionsCont& oc, NBNodeCont& nc) {
79  const GeoConvHelper& gch = GeoConvHelper::getFinal();
80  bool useGeo = oc.exists("proj.plain-geo") && oc.getBool("proj.plain-geo");
81  if (useGeo && !gch.usingGeoProjection()) {
82  WRITE_WARNING("Ignoring option \"proj.plain-geo\" because no geo-conversion has been defined");
83  useGeo = false;
84  }
85  const bool geoAccuracy = useGeo || gch.usingInverseGeoProjection();
86 
87  OutputDevice& device = OutputDevice::getDevice(oc.getString("plain-output-prefix") + ".nod.xml");
88  device.writeXMLHeader("nodes", NWFrame::MAJOR_VERSION + " xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xsi:noNamespaceSchemaLocation=\"http://sumo.sf.net/xsd/nodes_file.xsd\"");
89 
90  // write network offsets and projection to allow reconstruction of original coordinates
91  if (!useGeo) {
92  NWWriter_SUMO::writeLocation(device);
93  }
94 
95  // write nodes
96  for (std::map<std::string, NBNode*>::const_iterator i = nc.begin(); i != nc.end(); ++i) {
97  NBNode* n = (*i).second;
98  device.openTag(SUMO_TAG_NODE);
99  device.writeAttr(SUMO_ATTR_ID, n->getID());
100  // write position
101  Position pos = n->getPosition();
102  if (useGeo) {
103  gch.cartesian2geo(pos);
104  }
105  if (geoAccuracy) {
106  device.setPrecision(GEO_OUTPUT_ACCURACY);
107  }
108  NWFrame::writePositionLong(pos, device);
109  if (geoAccuracy) {
110  device.setPrecision();
111  }
112 
113  device.writeAttr(SUMO_ATTR_TYPE, toString(n->getType()));
114  if (n->isTLControlled()) {
115  const std::set<NBTrafficLightDefinition*>& tlss = n->getControllingTLS();
116  // set may contain multiple programs for the same id.
117  // make sure ids are unique and sorted
118  std::set<std::string> tlsIDs;
119  for (std::set<NBTrafficLightDefinition*>::const_iterator it_tl = tlss.begin(); it_tl != tlss.end(); it_tl++) {
120  tlsIDs.insert((*it_tl)->getID());
121  }
122  std::vector<std::string> sortedIDs(tlsIDs.begin(), tlsIDs.end());
123  sort(sortedIDs.begin(), sortedIDs.end());
124  device.writeAttr(SUMO_ATTR_TLID, sortedIDs);
125  }
126  device.closeTag();
127  }
128  device.close();
129 }

132 void
133 NWWriter_XML::writeEdgesAndConnections(const OptionsCont& oc, NBNodeCont& nc, NBEdgeCont& ec) {
134  const GeoConvHelper& gch = GeoConvHelper::getFinal();
135  bool useGeo = oc.exists("proj.plain-geo") && oc.getBool("proj.plain-geo");
136  const bool geoAccuracy = useGeo || gch.usingInverseGeoProjection();
137 
138  OutputDevice& edevice = OutputDevice::getDevice(oc.getString("plain-output-prefix") + ".edg.xml");
139  edevice.writeXMLHeader("edges", NWFrame::MAJOR_VERSION + " xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xsi:noNamespaceSchemaLocation=\"http://sumo.sf.net/xsd/edges_file.xsd\"");
140  OutputDevice& cdevice = OutputDevice::getDevice(oc.getString("plain-output-prefix") + ".con.xml");
141  cdevice.writeXMLHeader("connections", NWFrame::MAJOR_VERSION + " xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xsi:noNamespaceSchemaLocation=\"http://sumo.sf.net/xsd/connections_file.xsd\"");
142  bool noNames = !oc.getBool("output.street-names");
143  for (std::map<std::string, NBEdge*>::const_iterator i = ec.begin(); i != ec.end(); ++i) {
144  // write the edge itself to the edges-files
145  NBEdge* e = (*i).second;
146  edevice.openTag(SUMO_TAG_EDGE);
147  edevice.writeAttr(SUMO_ATTR_ID, e->getID());
148  edevice.writeAttr(SUMO_ATTR_FROM, e->getFromNode()->getID());
149  edevice.writeAttr(SUMO_ATTR_TO, e->getToNode()->getID());
150  if (!noNames && e->getStreetName() != "") {
151  edevice.writeAttr(SUMO_ATTR_NAME, StringUtils::escapeXML(e->getStreetName()));
152  }
153  edevice.writeAttr(SUMO_ATTR_PRIORITY, e->getPriority());
154  // write the type if given
155  if (e->getTypeID() != "") {
156  edevice.writeAttr(SUMO_ATTR_TYPE, e->getTypeID());
157  }
158  edevice.writeAttr(SUMO_ATTR_NUMLANES, e->getNumLanes());
159  if (!e->hasLaneSpecificSpeed()) {
160  edevice.writeAttr(SUMO_ATTR_SPEED, e->getSpeed());
161  }
162  // write non-default geometry
163  if (!e->hasDefaultGeometry()) {
164  PositionVector geom = e->getGeometry();
165  if (useGeo) {
166  for (int i = 0; i < (int) geom.size(); i++) {
167  gch.cartesian2geo(geom[i]);
168  }
169  }
170  if (geoAccuracy) {
171  edevice.setPrecision(GEO_OUTPUT_ACCURACY);
172  }
173  edevice.writeAttr(SUMO_ATTR_SHAPE, geom);
174  if (geoAccuracy) {
175  edevice.setPrecision();
176  }
177  }
178  // write the spread type if not default ("right")
179  if (e->getLaneSpreadFunction() != LANESPREAD_RIGHT) {
180  edevice.writeAttr(SUMO_ATTR_SPREADTYPE, toString(e->getLaneSpreadFunction()));
181  }
182  // write the length if it was specified
183  if (e->hasLoadedLength()) {
184  edevice.writeAttr(SUMO_ATTR_LENGTH, e->getLoadedLength());
185  }
186  // some attributes can be set by edge default or per lane. Write as default if possible (efficiency)
187  if (e->getLaneWidth() != NBEdge::UNSPECIFIED_WIDTH && !e->hasLaneSpecificWidth()) {
188  edevice.writeAttr(SUMO_ATTR_WIDTH, e->getLaneWidth());
189  }
190  if (e->getOffset() != NBEdge::UNSPECIFIED_OFFSET && !e->hasLaneSpecificOffset()) {
191  edevice.writeAttr(SUMO_ATTR_OFFSET, e->getOffset());
192  }
193  if (!e->needsLaneSpecificOutput()) {
194  edevice.closeTag();
195  } else {
196  for (unsigned int i = 0; i < e->getLanes().size(); ++i) {
197  const NBEdge::Lane& lane = e->getLanes()[i];
198  edevice.openTag(SUMO_TAG_LANE);
199  edevice.writeAttr(SUMO_ATTR_INDEX, i);
200  // write allowed lanes
201  NWWriter_SUMO::writePermissions(edevice, lane.permissions);
202  NWWriter_SUMO::writePreferences(edevice, lane.preferred);
203  // write other attributes
204  if (lane.width != NBEdge::UNSPECIFIED_WIDTH && e->hasLaneSpecificWidth()) {
205  edevice.writeAttr(SUMO_ATTR_WIDTH, lane.width);
206  }
207  if (lane.offset != NBEdge::UNSPECIFIED_OFFSET && e->hasLaneSpecificOffset()) {
208  edevice.writeAttr(SUMO_ATTR_OFFSET, lane.offset);
209  }
210  if (e->hasLaneSpecificSpeed()) {
211  edevice.writeAttr(SUMO_ATTR_SPEED, lane.speed);
212  }
213  edevice.closeTag();
214  }
215  edevice.closeTag();
216  }
217  // write this edge's connections to the connections-files
218  e->sortOutgoingConnectionsByIndex();
219  const std::vector<NBEdge::Connection> connections = e->getConnections();
220  for (std::vector<NBEdge::Connection>::const_iterator c = connections.begin(); c != connections.end(); ++c) {
221  NWWriter_SUMO::writeConnection(cdevice, *e, *c, false, NWWriter_SUMO::PLAIN);
222  }
223  if (connections.size() > 0) {
224  cdevice << "\n";
225  }
226  }
227 
228  // write loaded prohibitions to the connections-file
229  for (std::map<std::string, NBNode*>::const_iterator i = nc.begin(); i != nc.end(); ++i) {
230  NWWriter_SUMO::writeProhibitions(cdevice, i->second->getProhibitions());
231  }
232  edevice.close();
233  cdevice.close();
234 }

237 void
238 NWWriter_XML::writeTrafficLights(const OptionsCont& oc, NBTrafficLightLogicCont& tc, NBEdgeCont& ec) {
239  OutputDevice& device = OutputDevice::getDevice(oc.getString("plain-output-prefix") + ".tll.xml");
240  device.writeXMLHeader("tlLogics", NWFrame::MAJOR_VERSION + " xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xsi:noNamespaceSchemaLocation=\"http://sumo.sf.net/xsd/tllogic_file.xsd\"");
241  NWWriter_SUMO::writeTrafficLights(device, tc);
242  // we also need to remember the associations between tlLogics and connections
243  // since the information in con.xml is insufficient
244  for (std::map<std::string, NBEdge*>::const_iterator i = ec.begin(); i != ec.end(); ++i) {
245  NBEdge* e = (*i).second;
246  // write this edge's tl-controlled connections
247  const std::vector<NBEdge::Connection> connections = e->getConnections();
248  for (std::vector<NBEdge::Connection>::const_iterator c = connections.begin(); c != connections.end(); ++c) {
249  if (c->tlID != "") {
250  NWWriter_SUMO::writeConnection(device, *e, *c, false, NWWriter_SUMO::TLL);
251  }
252  }
253  }
254  device.close();
255 }

258 void
259 NWWriter_XML::writeJoinedJunctions(const OptionsCont& oc, NBNodeCont& nc) {
260  OutputDevice& device = OutputDevice::getDevice(oc.getString("junctions.join-output"));
261  device.writeXMLHeader("nodes", NWFrame::MAJOR_VERSION + " xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xsi:noNamespaceSchemaLocation=\"http://sumo.sf.net/xsd/nodes_file.xsd\"");
262  const std::vector<std::set<std::string> >& clusters = nc.getJoinedClusters();
263  for (std::vector<std::set<std::string> >::const_iterator it = clusters.begin(); it != clusters.end(); it++) {
264  assert((*it).size() > 0);
265  device.openTag(SUMO_TAG_JOIN);
266  // prepare string
267  std::ostringstream oss;
268  for (std::set<std::string>::const_iterator it_id = it->begin(); it_id != it->end(); it_id++) {
269  oss << *it_id << " ";
270  }
271  // remove final space
272  std::string ids = oss.str();
273  device.writeAttr(SUMO_ATTR_NODES, ids.substr(0, ids.size() - 1));
274  device.closeTag();
275  }
276  device.close();
277 }

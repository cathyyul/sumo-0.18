102 void
103 GUIMessageWindow::appendText(GUIEventType eType, const std::string& msg) {
104  if (!isEnabled()) {
105  show();
106  }
107  // build the styled message
108  FXint style = 1;
109  switch (eType) {
110  case EVENT_ERROR_OCCURED:
111  // color: red
112  style = 2;
113  break;
114  case EVENT_WARNING_OCCURED:
115  // color: yellow
116  style = 3;
117  break;
118  case EVENT_MESSAGE_OCCURED:
119  // color: green
120  style = 1;
121  break;
122  default:
123  assert(false);
124  }
125  // insert message to buffer
126  FXText::appendStyledText(msg.c_str(), (FXint) msg.length(), style + 1, true);
127  FXText::setCursorPos(getLength() - 1);
128  FXText::setBottomLine(getLength() - 1);
129  if (isEnabled()) {
130  layout();
131  update();
132  }
133 }

162 void
163 GUIMessageWindow::registerMsgHandlers() {
164  if (myMessageRetriever == 0) {
165  // initialize only if registration is requested
166  myMessageRetriever = new MsgOutputDevice(this, EVENT_MESSAGE_OCCURED);
167  myErrorRetriever = new MsgOutputDevice(this, EVENT_ERROR_OCCURED);
168  myWarningRetriever = new MsgOutputDevice(this, EVENT_WARNING_OCCURED);
169  }
170  MsgHandler::getMessageInstance()->addRetriever(myMessageRetriever);
171  MsgHandler::getErrorInstance()->addRetriever(myErrorRetriever);
172  MsgHandler::getWarningInstance()->addRetriever(myWarningRetriever);
173 }

54 MSFCDExport::write(OutputDevice& of, SUMOTime timestep) {
55  const bool useGeo = OptionsCont::getOptions().getBool("fcd-output.geo");
56  const bool signals = OptionsCont::getOptions().getBool("fcd-output.signals");
57  MSVehicleControl& vc = MSNet::getInstance()->getVehicleControl();
58  MSVehicleControl::constVehIt it = vc.loadedVehBegin();
59  MSVehicleControl::constVehIt end = vc.loadedVehEnd();
60 
61  of.openTag("timestep").writeAttr(SUMO_ATTR_TIME, time2string(timestep));
62  for (; it != end; ++it) {
63  const MSVehicle* veh = static_cast<const MSVehicle*>((*it).second);
64  if (veh->isOnRoad()) {
65  Position pos = veh->getPosition();
66  MSLane* lane = veh->getLane();
67  if (useGeo) {
68  of.setPrecision(GEO_OUTPUT_ACCURACY);
69  GeoConvHelper::getFinal().cartesian2geo(pos);
70  }
71  of.openTag("vehicle");
72  of.writeAttr(SUMO_ATTR_ID, veh->getID());
73  of.writeAttr(SUMO_ATTR_X, pos.x());
74  of.writeAttr(SUMO_ATTR_Y, pos.y());
75  of.writeAttr(SUMO_ATTR_ANGLE, veh->getAngle());
76  of.writeAttr(SUMO_ATTR_TYPE, veh->getVehicleType().getID());
77  of.writeAttr(SUMO_ATTR_SPEED, veh->getSpeed());
78  of.writeAttr(SUMO_ATTR_POSITION, veh->getPositionOnLane());
79  of.writeAttr(SUMO_ATTR_LANE, lane->getID());
80  of.writeAttr(SUMO_ATTR_SLOPE, lane->getShape().slopeDegreeAtOffset(veh->getPositionOnLane()));
81  if (signals) {
82  of.writeAttr("signals", toString(veh->getSignals()));
83  }
84  of.closeTag();
85  }
86  }
87  of.closeTag();
88 }

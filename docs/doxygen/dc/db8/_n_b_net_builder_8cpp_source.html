77 void
78 NBNetBuilder::applyOptions(OptionsCont& oc) {
79  // we possibly have to load the edges to keep
80  if (oc.isSet("keep-edges.input-file")) {
81  std::ifstream strm(oc.getString("keep-edges.input-file").c_str());
82  if (!strm.good()) {
83  throw ProcessError("Could not load names of edges too keep from '" + oc.getString("keep-edges.input-file") + "'.");
84  }
85  std::ostringstream oss;
86  bool first = true;
87  while (strm.good()) {
88  if (!first) {
89  oss << ',';
90  }
91  std::string name;
92  strm >> name;
93  oss << name;
94  first = false;
95  }
96  oc.set("keep-edges.explicit", oss.str());
97  }
98  // apply options to type control
99  myTypeCont.setDefaults(oc.getInt("default.lanenumber"), oc.getFloat("default.speed"), oc.getInt("default.priority"));
100  // apply options to edge control
101  myEdgeCont.applyOptions(oc);
102  // apply options to traffic light logics control
103  myTLLCont.applyOptions(oc);
104 }

107 void
108 NBNetBuilder::compute(OptionsCont& oc,
109  const std::set<std::string>& explicitTurnarounds,
110  bool removeUnwishedNodes) {
111  GeoConvHelper& geoConvHelper = GeoConvHelper::getProcessing();
112 
113 
114  // MODIFYING THE SETS OF NODES AND EDGES
115 
116  // Removes edges that are connecting the same node
117  PROGRESS_BEGIN_MESSAGE("Removing self-loops");
118  myNodeCont.removeSelfLoops(myDistrictCont, myEdgeCont, myTLLCont);
119  PROGRESS_DONE_MESSAGE();
120  //
121  if (oc.exists("remove-edges.isolated") && oc.getBool("remove-edges.isolated")) {
122  PROGRESS_BEGIN_MESSAGE("Finding isolated roads");
123  myNodeCont.removeIsolatedRoads(myDistrictCont, myEdgeCont, myTLLCont);
124  PROGRESS_DONE_MESSAGE();
125  }
126  //
127  if (oc.exists("keep-edges.postload") && oc.getBool("keep-edges.postload")) {
128  if (oc.isSet("keep-edges.explicit")) {
129  PROGRESS_BEGIN_MESSAGE("Removing unwished edges");
130  myEdgeCont.removeUnwishedEdges(myDistrictCont);
131  PROGRESS_DONE_MESSAGE();
132  }
133  }
134  // join junctions (may create new "geometry"-nodes so it needs to come before removing these
135  if (oc.exists("junctions.join-exclude") && oc.isSet("junctions.join-exclude")) {
136  myNodeCont.addJoinExclusion(oc.getStringVector("junctions.join-exclude"));
137  }
138  unsigned int numJoined = myNodeCont.joinLoadedClusters(myDistrictCont, myEdgeCont, myTLLCont);
139  if (oc.getBool("junctions.join")) {
140  PROGRESS_BEGIN_MESSAGE("Joining junction clusters");
141  // preliminary geometry computations to determine the length of edges
142  // This depends on turning directions and sorting of edge list
143  // in case junctions are joined geometry computations have to be repeated
144  NBTurningDirectionsComputer::computeTurnDirections(myNodeCont);
145  NBNodesEdgesSorter::sortNodesEdges(myNodeCont, oc.getBool("lefthand"));
146  myNodeCont.computeNodeShapes(oc.getBool("lefthand"));
147  myEdgeCont.computeEdgeShapes();
148  // preliminary roundabout computations to avoid destroying roundabouts
149  if (oc.getBool("roundabouts.guess") || (oc.isDefault("roundabouts.guess") && myHaveSeenRoundabouts)) {
150  assert(myRoundabouts.size() == 0);
151  myEdgeCont.guessRoundabouts(myRoundabouts);
152  for (std::vector<EdgeVector>::const_iterator it_round = myRoundabouts.begin();
153  it_round != myRoundabouts.end(); ++it_round) {
154  std::vector<std::string> nodeIDs;
155  for (EdgeVector::const_iterator it_edge = it_round->begin(); it_edge != it_round->end(); ++it_edge) {
156  nodeIDs.push_back((*it_edge)->getToNode()->getID());
157  }
158  myNodeCont.addJoinExclusion(nodeIDs);
159  }
160  myRoundabouts.clear();
161  }
162  numJoined += myNodeCont.joinJunctions(oc.getFloat("junctions.join-dist"), myDistrictCont, myEdgeCont, myTLLCont);
163  // reset geometry to avoid influencing subsequent steps (ramps.guess)
164  myEdgeCont.computeLaneShapes();
165  PROGRESS_DONE_MESSAGE();
166  }
167  if (numJoined > 0) {
168  // bit of a misnomer since we're already done
169  WRITE_MESSAGE(" Joined " + toString(numJoined) + " junction cluster(s).");
170  }
171  //
172  if (removeUnwishedNodes) {
173  unsigned int no = 0;
174  const bool removeGeometryNodes = oc.exists("geometry.remove") && oc.getBool("geometry.remove");
175  PROGRESS_BEGIN_MESSAGE("Removing empty nodes" + std::string(removeGeometryNodes ? " and geometry nodes" : ""));
176  no = myNodeCont.removeUnwishedNodes(myDistrictCont, myEdgeCont, myJoinedEdges, myTLLCont, removeGeometryNodes);
177  PROGRESS_DONE_MESSAGE();
178  WRITE_MESSAGE(" " + toString(no) + " nodes removed.");
179  }
180  // MOVE TO ORIGIN
181  if (!oc.getBool("offset.disable-normalization") && oc.isDefault("offset.x") && oc.isDefault("offset.y")) {
182  moveToOrigin(geoConvHelper);
183  }
184  geoConvHelper.computeFinal(); // information needed for location element fixed at this point
185 
186  if (oc.exists("geometry.min-dist") && oc.isSet("geometry.min-dist")) {
187  PROGRESS_BEGIN_MESSAGE("Reducing geometries");
188  myEdgeCont.reduceGeometries(oc.getFloat("geometry.min-dist"));
189  PROGRESS_DONE_MESSAGE();
190  }
191  // @note: removing geometry can create similar edges so joinSimilarEdges must come afterwards
192  // @note: likewise splitting can destroy similarities so joinSimilarEdges must come before
193  PROGRESS_BEGIN_MESSAGE("Joining similar edges");
194  myJoinedEdges.init(myEdgeCont);
195  myNodeCont.joinSimilarEdges(myDistrictCont, myEdgeCont, myTLLCont);
196  PROGRESS_DONE_MESSAGE();
197  //
198  if (oc.exists("geometry.split") && oc.getBool("geometry.split")) {
199  PROGRESS_BEGIN_MESSAGE("Splitting geometry edges");
200  myEdgeCont.splitGeometry(myNodeCont);
201  PROGRESS_DONE_MESSAGE();
202  }
203  // guess ramps
204  if ((oc.exists("ramps.guess") && oc.getBool("ramps.guess")) || (oc.exists("ramps.set") && oc.isSet("ramps.set"))) {
205  PROGRESS_BEGIN_MESSAGE("Guessing and setting on-/off-ramps");
206  NBNodesEdgesSorter::sortNodesEdges(myNodeCont, oc.getBool("lefthand"));
207  NBRampsComputer::computeRamps(*this, oc);
208  PROGRESS_DONE_MESSAGE();
209  }
210 
211  // check whether any not previously setable connections may be set now
212  myEdgeCont.recheckPostProcessConnections();
213  //
214  if (oc.exists("geometry.max-angle")) {
215  myEdgeCont.checkGeometries(
216  oc.getFloat("geometry.max-angle"),
217  oc.getFloat("geometry.min-radius"),
218  oc.getBool("geometry.min-radius.fix"));
219  }
220  //
221  myEdgeCont.computeLaneShapes();
222 
223  // APPLY SPEED MODIFICATIONS
224  if (oc.exists("speed.offset")) {
225  const SUMOReal speedOffset = oc.getFloat("speed.offset");
226  const SUMOReal speedFactor = oc.getFloat("speed.factor");
227  if (speedOffset != 0 || speedFactor != 1) {
228  PROGRESS_BEGIN_MESSAGE("Applying speed modifications");
229  for (std::map<std::string, NBEdge*>::const_iterator i = myEdgeCont.begin(); i != myEdgeCont.end(); ++i) {
230  (*i).second->setSpeed(-1, (*i).second->getSpeed() * speedFactor + speedOffset);
231  }
232  PROGRESS_DONE_MESSAGE();
233  }
234  }
235 
236  // CONNECTIONS COMPUTATION
237  //
238  PROGRESS_BEGIN_MESSAGE("Computing turning directions");
239  NBTurningDirectionsComputer::computeTurnDirections(myNodeCont);
240  PROGRESS_DONE_MESSAGE();
241  //
242  PROGRESS_BEGIN_MESSAGE("Sorting nodes' edges");
243  NBNodesEdgesSorter::sortNodesEdges(myNodeCont, oc.getBool("lefthand"));
244  PROGRESS_DONE_MESSAGE();
245  //
246  PROGRESS_BEGIN_MESSAGE("Computing node types");
247  NBNodeTypeComputer::computeNodeTypes(myNodeCont);
248  PROGRESS_DONE_MESSAGE();
249  //
250  PROGRESS_BEGIN_MESSAGE("Computing priorities");
251  NBEdgePriorityComputer::computeEdgePriorities(myNodeCont);
252  PROGRESS_DONE_MESSAGE();
253  //
254  PROGRESS_BEGIN_MESSAGE("Computing approached edges");
255  myEdgeCont.computeEdge2Edges(oc.getBool("no-left-connections"));
256  PROGRESS_DONE_MESSAGE();
257  //
258  if (oc.getBool("roundabouts.guess") || (oc.isDefault("roundabouts.guess") && myHaveSeenRoundabouts)) {
259  PROGRESS_BEGIN_MESSAGE("Guessing and setting roundabouts");
260  myEdgeCont.guessRoundabouts(myRoundabouts);
261  PROGRESS_DONE_MESSAGE();
262  }
263  //
264  PROGRESS_BEGIN_MESSAGE("Computing approaching lanes");
265  myEdgeCont.computeLanes2Edges();
266  PROGRESS_DONE_MESSAGE();
267  //
268  PROGRESS_BEGIN_MESSAGE("Dividing of lanes on approached lanes");
269  myNodeCont.computeLanes2Lanes();
270  myEdgeCont.sortOutgoingLanesConnections();
271  PROGRESS_DONE_MESSAGE();
272  //
273  PROGRESS_BEGIN_MESSAGE("Processing turnarounds");
274  if (!oc.getBool("no-turnarounds")) {
275  myEdgeCont.appendTurnarounds(oc.getBool("no-turnarounds.tls"));
276  } else {
277  myEdgeCont.appendTurnarounds(explicitTurnarounds, oc.getBool("no-turnarounds.tls"));
278  }
279  PROGRESS_DONE_MESSAGE();
280  //
281  PROGRESS_BEGIN_MESSAGE("Rechecking of lane endings");
282  myEdgeCont.recheckLanes();
283  PROGRESS_DONE_MESSAGE();
284 
285 
286  // GEOMETRY COMPUTATION
287  //
288  PROGRESS_BEGIN_MESSAGE("Computing node shapes");
289  myNodeCont.computeNodeShapes(oc.getBool("lefthand"));
290  PROGRESS_DONE_MESSAGE();
291  //
292  PROGRESS_BEGIN_MESSAGE("Computing edge shapes");
293  myEdgeCont.computeEdgeShapes();
294  PROGRESS_DONE_MESSAGE();
295 
296 
297  // GUESS TLS POSITIONS
298  PROGRESS_BEGIN_MESSAGE("Assigning nodes to traffic lights");
299  if (oc.isSet("tls.set")) {
300  std::vector<std::string> tlControlledNodes = oc.getStringVector("tls.set");
301  TrafficLightType type = SUMOXMLDefinitions::TrafficLightTypes.get(oc.getString("tls.default-type"));
302  for (std::vector<std::string>::const_iterator i = tlControlledNodes.begin(); i != tlControlledNodes.end(); ++i) {
303  NBNode* node = myNodeCont.retrieve(*i);
304  if (node == 0) {
305  WRITE_WARNING("Building a tl-logic for node '" + *i + "' is not possible." + "\n The node '" + *i + "' is not known.");
306  } else {
307  myNodeCont.setAsTLControlled(node, myTLLCont, type);
308  }
309  }
310  }
311  myNodeCont.guessTLs(oc, myTLLCont);
312  PROGRESS_DONE_MESSAGE();
313  //
314  if (oc.getBool("tls.join")) {
315  PROGRESS_BEGIN_MESSAGE("Joining traffic light nodes");
316  myNodeCont.joinTLS(myTLLCont, oc.getFloat("tls.join-dist"));
317  PROGRESS_DONE_MESSAGE();
318  }
319 
320 
321  // COMPUTING RIGHT-OF-WAY AND TRAFFIC LIGHT PROGRAMS
322  //
323  PROGRESS_BEGIN_MESSAGE("Computing traffic light control information");
324  myTLLCont.setTLControllingInformation(myEdgeCont);
325  PROGRESS_DONE_MESSAGE();
326  //
327  PROGRESS_BEGIN_MESSAGE("Computing node logics");
328  myNodeCont.computeLogics(myEdgeCont, oc);
329  PROGRESS_DONE_MESSAGE();
330  //
331  PROGRESS_BEGIN_MESSAGE("Computing traffic light logics");
332  std::pair<unsigned int, unsigned int> numbers = myTLLCont.computeLogics(myEdgeCont, oc);
333  PROGRESS_DONE_MESSAGE();
334  std::string progCount = "";
335  if (numbers.first != numbers.second) {
336  progCount = "(" + toString(numbers.second) + " programs) ";
337  }
338  WRITE_MESSAGE(" " + toString(numbers.first) + " traffic light(s) " + progCount + "computed.");
339  //
340  if (oc.isSet("street-sign-output")) {
341  PROGRESS_BEGIN_MESSAGE("Generating street signs");
342  myEdgeCont.generateStreetSigns();
343  PROGRESS_DONE_MESSAGE();
344  }
345 
346  // FINISHING INNER EDGES
347  if (!oc.getBool("no-internal-links")) {
348  PROGRESS_BEGIN_MESSAGE("Building inner edges");
349  for (std::map<std::string, NBEdge*>::const_iterator i = myEdgeCont.begin(); i != myEdgeCont.end(); ++i) {
350  (*i).second->sortOutgoingConnectionsByIndex();
351  }
352  for (std::map<std::string, NBNode*>::const_iterator i = myNodeCont.begin(); i != myNodeCont.end(); ++i) {
353  (*i).second->buildInnerEdges();
354  }
355  PROGRESS_DONE_MESSAGE();
356  }
357 
358 
359  // report
360  WRITE_MESSAGE("-----------------------------------------------------");
361  WRITE_MESSAGE("Summary:");
362  myNodeCont.printBuiltNodesStatistics();
363  WRITE_MESSAGE(" Network boundaries:");
364  WRITE_MESSAGE(" Original boundary : " + toString(geoConvHelper.getOrigBoundary()));
365  WRITE_MESSAGE(" Applied offset : " + toString(geoConvHelper.getOffsetBase()));
366  WRITE_MESSAGE(" Converted boundary : " + toString(geoConvHelper.getConvBoundary()));
367  WRITE_MESSAGE("-----------------------------------------------------");
368  NBRequest::reportWarnings();

372  WRITE_WARNING("Network contains very large coordinates and will probably flicker in the GUI. Check for outlying nodes and make sure the network is shifted to the coordinate origin");

377 void
378 NBNetBuilder::moveToOrigin(GeoConvHelper& geoConvHelper) {
379  PROGRESS_BEGIN_MESSAGE("Moving network to origin");
380  // compute new boundary after network modifications have taken place
381  Boundary boundary;
382  for (std::map<std::string, NBNode*>::const_iterator it = myNodeCont.begin(); it != myNodeCont.end(); ++it) {
383  boundary.add(it->second->getPosition());
384  }
385  for (std::map<std::string, NBEdge*>::const_iterator it = myEdgeCont.begin(); it != myEdgeCont.end(); ++it) {
386  boundary.add(it->second->getGeometry().getBoxBoundary());
387  }
388  geoConvHelper.setConvBoundary(boundary);
389  const SUMOReal x = -boundary.xmin();
390  const SUMOReal y = -boundary.ymin();
391  for (std::map<std::string, NBNode*>::const_iterator i = myNodeCont.begin(); i != myNodeCont.end(); ++i) {
392  (*i).second->reshiftPosition(x, y);
393  }
394  for (std::map<std::string, NBEdge*>::const_iterator i = myEdgeCont.begin(); i != myEdgeCont.end(); ++i) {
395  (*i).second->reshiftPosition(x, y);
396  }
397  for (std::map<std::string, NBDistrict*>::const_iterator i = myDistrictCont.begin(); i != myDistrictCont.end(); ++i) {
398  (*i).second->reshiftPosition(x, y);
399  }
400  geoConvHelper.moveConvertedBy(x, y);
401  PROGRESS_DONE_MESSAGE();
402 }

406 NBNetBuilder::transformCoordinates(Position& from, bool includeInBoundary, GeoConvHelper* from_srs) {
407  Position orig(from);
408  bool ok = GeoConvHelper::getProcessing().x2cartesian(from, includeInBoundary);
409 #ifdef HAVE_INTERNAL
410  if (ok) {
411  const HeightMapper& hm = HeightMapper::get();
412  if (hm.ready()) {
413  if (from_srs != 0 && from_srs->usingGeoProjection()) {
414  from_srs->cartesian2geo(orig);
415  }
416  SUMOReal z = hm.getZ(orig);
417  from = Position(from.x(), from.y(), z);
418  }
419  }
420 #else
421  UNUSED_PARAMETER(from_srs);
422 #endif
423  return ok;
424 }

427 bool
428 NBNetBuilder::transformCoordinates(PositionVector& from, bool includeInBoundary, GeoConvHelper* from_srs) {
429  const SUMOReal maxLength = OptionsCont::getOptions().getFloat("geometry.max-segment-length");
430  if (maxLength > 0 && from.size() > 1) {
431  // transformation to cartesian coordinates must happen before we can check segment length
432  PositionVector copy = from;
433  for (int i = 0; i < (int) from.size(); i++) {
434  transformCoordinates(copy[i], false);
435  }
436  // check lengths and insert new points where needed (in the original
437  // coordinate system)
438  int inserted = 0;
439  for (int i = 0; i < (int)copy.size() - 1; i++) {
440  Position start = from[i + inserted];
441  Position end = from[i + inserted + 1];
442  SUMOReal length = copy[i].distanceTo(copy[i + 1]);
443  const Position step = (end - start) * (maxLength / length);
444  int steps = 0;
445  while (length > maxLength) {
446  length -= maxLength;
447  steps++;
448  from.insertAt(i + inserted + 1, start + (step * steps));
449  inserted++;
450  }
451  }
452  // now perform the transformation again so that height mapping can be
453  // performed for the new points
454  }
455  bool ok = true;
456  for (int i = 0; i < (int) from.size(); i++) {
457  ok = ok && transformCoordinates(from[i], includeInBoundary, from_srs);
458  }
459  return ok;
